def relax_custom_structure(
    calc_dir: pathlib.Path,
    casm_structure: xtal.Structure,
    calculator: typing.Any,
    fmax: float,
):
    calc_dir.mkdir(parents=True, exist_ok=True)

    structure_path = calc_dir / "structure.json"
    structure_with_properties_path = calc_dir / "structure_with_properties.json"
    status_path = calc_dir / "status.json"

    data = read_optional(status_path, default=dict())
    status = data.get("status", "none")

    if status == "complete":
        data = read_required(structure_with_properties_path)
        return xtal.Structure.from_dict(data)
    
    
    print("In working directory:", printpathstr(calc_dir))
    with working_dir(calc_dir):

        # Write structure.json
        safe_dump(casm_structure.to_dict(), path=structure_path, force=True)

        from datetime import datetime

        # Get the current datetime
        now = datetime.now()
        
        # Convert to ISO 8601 string
        iso_string = status_path
        print(f"ISO formatted datetime: {iso_string}")

        # Write status.json
        status_data = {
            "status": "setup",
            "starttime": datetime.now().isoformat(),
        }
        safe_dump(status_data, path=status_path, force=True)
        
        # Relax structure
        casm_structure_with_properties, Epot = relax(
            casm_structure=casm_structure,
            calculator=calculator,
            fmax=fmax,
        )
        
        # Write structure_with_properties.json
        safe_dump(
            casm_structure_with_properties.to_dict(),
            path=structure_with_properties_path,
            force=True,
        )    

        # Write status.json
        status_data["status"] = "complete"
        status_data["stoptime"] = datetime.now().isoformat()
        safe_dump(status_data, path=status_path, force=True)

    return casm_structure_with_properties


def Omega_lattice(
    r: typing.Optional[float] = None,
    a: typing.Optional[float] = None,
    c: typing.Optional[float] = None,
):
    """Create the omega structure lattice"""
    if r is not None:
        # from a_omega = bcc face diagonal,
        # a_omega = a_bcc * math.sqrt(2)
        a = math.sqrt(32/3) * r

        # from volume conservation:
        c = math.sqrt(8) * r
    return xtal_lattices.hexagonal(a=a, c=c)

def Omega_structure(
    r: typing.Optional[float] = None,
    a: typing.Optional[float] = None,
    c: typing.Optional[float] = None,
    z: float = 0.0,
    atom_type: typing.Union[str,list[str]] = "A",
    atom_properties: dict[str, np.ndarray] = {},
    global_properties: dict[str, np.ndarray] = {},
    conventional=False,
) -> xtal.Structure:
    """Create an omega structure, ideal or shuffled"""

    if isinstance(atom_type, str):
        atom_type = [atom_type, atom_type, atom_type]
        
    return xtal.Structure(
        lattice=Omega_lattice(r=r, a=a, c=c),
        atom_coordinate_frac=np.array(
            [
                [0.0, 0.0, 0.0],
                [1.0 / 3.0, 2.0 / 3.0, 0.5 + z],
                [2.0 / 3.0, 1.0 / 3.0, 0.5 - z],
            ]
        ).transpose(),
        atom_type=atom_type,
        atom_properties=atom_properties,
        global_properties=global_properties,
    )
