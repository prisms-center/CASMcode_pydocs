    "import math\n",
    "import typing\n",

    "def relax_custom_structure(\n",
    "    calc_dir: pathlib.Path,\n",
    "    casm_structure: xtal.Structure,\n",
    "    calculator: typing.Any,\n",
    "    fmax: float,\n",
    "):\n",
    "    calc_dir.mkdir(parents=True, exist_ok=True)\n",
    "\n",
    "    structure_path = calc_dir / \"structure.json\"\n",
    "    structure_with_properties_path = calc_dir / \"structure_with_properties.json\"\n",
    "    status_path = calc_dir / \"status.json\"\n",
    "\n",
    "    data = read_optional(status_path, default=dict())\n",
    "    status = data.get(\"status\", \"none\")\n",
    "\n",
    "    if status == \"complete\":\n",
    "        data = read_required(structure_with_properties_path)\n",
    "        return xtal.Structure.from_dict(data)\n",
    "    \n",
    "    \n",
    "    print(\"In working directory:\", printpathstr(calc_dir))\n",
    "    with working_dir(calc_dir):\n",
    "\n",
    "        # Write structure.json\n",
    "        safe_dump(casm_structure.to_dict(), path=structure_path, force=True)\n",
    "\n",
    "        from datetime import datetime\n",
    "\n",
    "        # Get the current datetime\n",
    "        now = datetime.now()\n",
    "        \n",
    "        # Convert to ISO 8601 string\n",
    "        iso_string = status_path\n",
    "        print(f\"ISO formatted datetime: {iso_string}\")\n",
    "\n",
    "        # Write status.json\n",
    "        status_data = {\n",
    "            \"status\": \"setup\",\n",
    "            \"starttime\": datetime.now().isoformat(),\n",
    "        }\n",
    "        safe_dump(status_data, path=status_path, force=True)\n",
    "        \n",
    "        # Relax structure\n",
    "        casm_structure_with_properties, Epot = relax(\n",
    "            casm_structure=casm_structure,\n",
    "            calculator=calculator,\n",
    "            fmax=fmax,\n",
    "        )\n",
    "        \n",
    "        # Write structure_with_properties.json\n",
    "        safe_dump(\n",
    "            casm_structure_with_properties.to_dict(),\n",
    "            path=structure_with_properties_path,\n",
    "            force=True,\n",
    "        )    \n",
    "\n",
    "        # Write status.json\n",
    "        status_data[\"status\"] = \"complete\"\n",
    "        status_data[\"stoptime\"] = datetime.now().isoformat()\n",
    "        safe_dump(status_data, path=status_path, force=True)\n",
    "\n",
    "    return casm_structure_with_properties\n",
    "\n",
    "\n",
    "def Omega_lattice(\n",
    "    r: typing.Optional[float] = None,\n",
    "    a: typing.Optional[float] = None,\n",
    "    c: typing.Optional[float] = None,\n",
    "):\n",
    "    \"\"\"Create the omega structure lattice\"\"\"\n",
    "    if r is not None:\n",
    "        # from a_omega = bcc face diagonal,\n",
    "        # a_omega = a_bcc * math.sqrt(2)\n",
    "        a = math.sqrt(32/3) * r\n",
    "\n",
    "        # from volume conservation:\n",
    "        c = math.sqrt(8) * r\n",
    "    return xtal_lattices.hexagonal(a=a, c=c)\n",
    "\n",
    "def Omega_structure(\n",
    "    r: typing.Optional[float] = None,\n",
    "    a: typing.Optional[float] = None,\n",
    "    c: typing.Optional[float] = None,\n",
    "    z: float = 0.0,\n",
    "    atom_type: typing.Union[str,list[str]] = \"A\",\n",
    "    atom_properties: dict[str, np.ndarray] = {},\n",
    "    global_properties: dict[str, np.ndarray] = {},\n",
    "    conventional=False,\n",
    ") -> xtal.Structure:\n",
    "    \"\"\"Create an omega structure, ideal or shuffled\"\"\"\n",
    "\n",
    "    if isinstance(atom_type, str):\n",
    "        atom_type = [atom_type, atom_type, atom_type]\n",
    "        \n",
    "    return xtal.Structure(\n",
    "        lattice=Omega_lattice(r=r, a=a, c=c),\n",
    "        atom_coordinate_frac=np.array(\n",
    "            [\n",
    "                [0.0, 0.0, 0.0],\n",
    "                [1.0 / 3.0, 2.0 / 3.0, 0.5 + z],\n",
    "                [2.0 / 3.0, 1.0 / 3.0, 0.5 - z],\n",
    "            ]\n",
    "        ).transpose(),\n",
    "        atom_type=atom_type,\n",
    "        atom_properties=atom_properties,\n",
    "        global_properties=global_properties,\n",
    "    )"
